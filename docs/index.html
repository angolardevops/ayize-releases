        type Sujeito {
    obs: list<dyn>
    fn subscreve(self, f) { self.obs.push(f) }
    fn notifica(self, ev) { for f in self.obs { f(ev) } }
}

fn main() {
    let s = Sujeito { obs: [] }
    s.subscreve(fn(e) => print("A viu: ${e}"))
    s.subscreve(fn(e) => print("B viu: ${e}"))
    s.notifica("evento!")              // os dois observadores reagem
}

        type Circulo : Forma  { r: float  fn area(self) -> float { return 3.14159 * self.r * self.r } }
type Quadrado : Forma { l: float  fn area(self) -> float { return self.l * self.l } }

fn cria_forma(tipo, tam) {
    if tipo == "circulo" { return Circulo { r: tam } }
    return Quadrado { l: tam }
}

fn main() {
    let f = cria_forma("circulo", 2.0)
    print(f.area())                    // 12.566...  (dispatch pelo contrato)
}

        fn parse_porta(s) {
    let n = s.to_int()?                // se Err, propaga já para o chamador
    if n < 0 { return Err("porta negativa") }
    return Ok(n)
}

fn main() {
    let r = parse_porta("8080")
    if is_ok(r) { print("porta = ${unwrap_or(r, 0)}") }
    else        { print("inválida") }
}

        import nn
import linalg

fn main() {
    let x = nn.param(linalg.matrix(4, 2, [1.0,1.0, 1.0,2.0, 1.0,3.0, 1.0,4.0])?)
    let t = linalg.matrix(4, 1, [3.0, 5.0, 7.0, 9.0])?    // alvo: y = 2x + 1
    let w = nn.param(linalg.matrix(2, 1, [0.0, 0.0])?)
    let opt = nn.sgd([w], 0.02)
    for step in 0..400 {
        opt.zero_grad()
        let loss = nn.mse(nn.matmul(x, w), t)
        nn.backward(loss)              // autograd reverse-mode
        opt.step()
    }
    print("W ≈ ${w.value().to_list()}")   // ≈ [1, 2]
}

        import gpt
import tokenizer

fn main() {
    let corpus = "ayize treina llms. ayize treina llms. ayize treina llms."
    let tok = tokenizer.char(corpus)
    let v = tok.vocab_size()
    let ids = tok.encode(corpus)

    let model = gpt.new(v, 32, ids.len(), 64, 4, 7)
    let r = model.train(ids, 300, 0.02)
    print("treino: loss ${r["first"]} → ${r["last"]}")

    let out = model.generate(tok.encode("ayize "), 22, 0.0, 0)
    print("gerado = ${tok.decode(out)}")
}

        import gpt

    // classificador: modelo-base + cabeça linear [d → num_classes]

}
